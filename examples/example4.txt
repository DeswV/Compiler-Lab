var m; // 普通变量
var *p; // 一级指针变量
var **q; // 二级指针变量
var ***r; // 三级指针变量
var * a[10] ; // 一维数组，元素为一级指针
begin
p := &m; // 取普通变量地址并赋值给一级指针变量
q := &p; // 取一级指针变量地址并赋值给二级指针变量
r := &q; // 取二级指针变量地址并赋值给三级指针变量
***r := 100; // 访问三级指针变量
a[0] := &m; // 给一维数组元素赋值
a[1] := p;
print(* a[0], *a[1], m); // 输出三个 100
end.
